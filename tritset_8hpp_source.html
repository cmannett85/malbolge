105  constexpr explicit tritset(T value = 0) :
106  v_{0}
107  {
108  auto q = value % (max+1);
109  for (auto i = 0u; i < width; ++i) {
110  v_ |= ((q % trit::base) & bmask) << (i*trit::bits_per_trit);
111  if (q < trit::base) {
112  break;
113  }
114  q /= trit::base;
115  }
116  }
