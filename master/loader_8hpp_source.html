76  // However the act of denormalising checks the syntax, so we can skip

79  } else {

82  for (auto it = first; it != last;) {
83  if (std::isspace(*it)) {
84  if (*it == '\n') {

87  } else {
88  ++loc.column;
89  }

91  last = algorithm::remove_from_range(it, last);
92  continue;
93  }
94 
95  auto instr = pre_cipher_instruction(*it, i);
96  if (!instr) {

101  }

106  loc};
107  }
108  ++loc.column;
109  ++it;
110  ++i;
111  }
