282  template <typename InputIt>
283  explicit virtual_memory(InputIt first, InputIt last) :
284  mem_{std::make_unique<base::element_type>()}
285  {
286  const auto program_length = std::distance(first, last);
287  if (program_length < 2) {
288  throw parse_exception{"Program data must be at least 2 characters"};
289  }
290 
291  if (static_cast<std::size_t>(program_length) > size()) {
292  throw parse_exception{"Program data must be less than "
293  "math::ternary::max"};
294  }
295 
296  // Copy the program data in
297  auto op_it = std::copy(first, last, mem_->begin());
298 
299  // Fill the remainder of the data space with the ternary op applied with
300  // the previous two addresses
301  for (; op_it != mem_->end(); ++op_it) {
302  *op_it = (op_it-1)->op(*(op_it-2));
303  }
304  }
