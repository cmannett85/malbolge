161  {
162  const auto positive = offset >= 0;
163  offset = std::abs(offset) % data_.size();
164  if (positive) {
165  const auto dist_to_end = std::distance(current_, data_.end());
166  if (offset > dist_to_end) {
167  offset -= dist_to_end;
168  current_ = data_.begin() + offset;

170  current_ += offset;

174  if (offset > dist_to_start) {

176  current_ = data_.end() - offset;
177  } else {
178  current_ -= offset;

288  template <typename InputIt>
289  explicit virtual_memory(InputIt first, InputIt last) :
290  mem_{std::make_unique<base::element_type>()}
291  {
292  const auto program_length = std::distance(first, last);
293  if (program_length < 2) {
294  throw parse_exception{"Program data must be at least 2 characters"};
295  }
296 
297  if (static_cast<std::size_t>(program_length) > size()) {
298  throw parse_exception{"Program data must be less than "
299  "math::ternary::max"};
300  }
301 
302  // Copy the program data in
303  auto op_it = std::copy(first, last, mem_->begin());
304 
305  // Fill the remainder of the data space with the ternary op applied with
306  // the previous two addresses
307  for (; op_it != mem_->end(); ++op_it) {
308  *op_it = (op_it-1)->op(*(op_it-2));
309  }
310  }
