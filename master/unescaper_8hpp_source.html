20 // These functions are derived from the escape list detailed here:
21 // https://stackoverflow.com/questions/10220401/rules-for-c-string-literals-escape-character
22 template <typename Iter>
23 [[nodiscard]]
24 std::optional<char> is_control_character(Iter it) noexcept
25 {
26  static const auto mapper = std::unordered_map<char, char>{
27  {'a', '\a'},
28  {'b', '\b'},
29  {'t', '\t'},
30  {'n', '\n'},
31  {'v', '\v'},
32  {'f', '\f'},
33  {'r', '\r'},
34  };
35  auto c_it = mapper.find(*it);
36  if (c_it != mapper.end()) {
37  return c_it->second;
38  }
39 
40  return {};
41 }

43 template <typename Iter>
44 [[nodiscard]]
45 std::optional<char> is_punctuation_character(Iter it) noexcept
46 {
47  static const auto mapper = std::array<char, 4>{'\"', '\'', '\?', '\\'};
48  if (std::find(mapper.begin(), mapper.end(), *it) != mapper.end()) {
49  return *it;
50  }
51 
52  return {};
53 }

55 // The octal and hex functions are 'greedy' i.e. they move the iterator along
56 template <typename Iter>
57 [[nodiscard]]
58 std::optional<char> is_octal_character(Iter& it, Iter end)
59 {
60  static constexpr auto base_8 = 8u;
61  static constexpr auto max_octal_digits = 3u;
62 
63  auto count = 0u;
64  for (auto c_it = it; (c_it != end) && (count < max_octal_digits); ++c_it) {
65  if ((*c_it >= '0') && (*c_it <= '7')) {
66  ++count;
67  } else {
68  break;
69  }
70  }
71  if (count > 0u) {
72  const auto str = std::string_view{&(*it), count};
73  const auto result = utility::from_chars_with_base<char>(str, base_8);
74  it += count - 1; // Move the iterator along to last octal character
75  return result;
76  }
77 
78  return {};
79 }

81 template <typename Iter>
82 [[nodiscard]]
83 std::optional<char> is_hex_character(Iter& it, Iter end)
84 {
85  static constexpr auto base_16 = 16u;
86 
87  if (*it != 'x') {
88  return {};
89  }
90 
91  // Skip the x
92  ++it;
93 
94  auto count = 0u;
95  for (auto c_it = it; c_it != end; ++c_it) {
96  if (std::isxdigit(*c_it)) {
97  ++count;
98  } else {
99  break;
100  }
101  }
102  if (count > 0u) {
103  const auto str = std::string_view{&(*it), count};
104  const auto result = utility::from_chars_with_base<char>(str, base_16);
105  it += count - 1; // Move the iterator along to last hex character
106  return result;
107  }
108 
109  return {};
110 }

130 template <typename Traits = std::char_traits<char>,
131  typename Allocator = std::allocator<char>>
132 [[nodiscard]]
133 std::basic_string<char, Traits, Allocator>
134 unescape_ascii(std::basic_string_view<char, Traits>& str)
135 {
136  auto unescaped = std::basic_string<char, Traits, Allocator>{};
137  unescaped.reserve(str.size());
138 
139  for (auto it = str.begin(); it != str.end(); ++it) {
140  if (*it == '\\') {
141  ++it;
142 
143  auto result = detail::is_control_character(it);
144  if (result) {
145  unescaped += *result;
146  continue;
147  }
148 
149  result = detail::is_punctuation_character(it);
150  if (result) {
151  unescaped += *result;
152  continue;
153  }
154 
155  result = detail::is_octal_character(it, str.end());
156  if (result) {
157  unescaped += *result;
158  continue;
159  }
160 
161  result = detail::is_hex_character(it, str.end());
162  if (result) {
163  unescaped += *result;
164  continue;
165  }
166 
167  throw std::invalid_argument{"Cannot parse escape character"};
168  } else {
169  unescaped.push_back(*it);
170  }
171  }
172 
173  return unescaped;
174 }
